# Option - Category

## Props

<Props of={EventOptionCategory} />

## Basic usage

<Playground>
  {() => {
    const [selectedCategory, setSelectedCategory] = React.useState(0);
    const categories = ['Møbler', 'tekstil', 'elekrontikk'];
    const [isEditing, setIsEditing] = React.useState(false);
    const onChange = (e) => {
      setSelectedCategory(e.currentTarget.value);
    };
    return (
      <>
        <button onClick={() => {setIsEditing(!isEditing)}}>Edit</button>
        <EventOptionCategory onChange={onChange} selectedCategory={selectedCategory} categories={categories} isEditing={isEditing} />
      </>
    )
  }}
</Playground>
